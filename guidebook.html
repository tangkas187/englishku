// audio.js – Signal envelope implementation
function playTone(frequency, duration, type, volume) {
  const oscillator = audioCtx.createOscillator();
  const gainNode   = audioCtx.createGain();
  oscillator.connect(gainNode);
  gainNode.connect(audioCtx.destination);

  oscillator.type = type; // 'sine' | 'triangle' | 'sawtooth'
  oscillator.frequency.setValueAtTime(frequency, audioCtx.currentTime);

  // Envelope: attack (0→volume) + decay (volume→0)
  gainNode.gain.setValueAtTime(0, audioCtx.currentTime);
  gainNode.gain.linearRampToValueAtTime(volume, audioCtx.currentTime + 0.01);
  gainNode.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
}

Tahap 2 – Run-Length Encoding (RLE) untuk teks

// multimedia.js – RLE encoder
function rleEncode(str) {
  let result = '', i = 0;
  while (i < str.length) {
    let count = 1;
    while (str[i + count] === str[i] && count < 255) count++;
    result += (count > 1 ? count : '') + str[i];
    i += count;
  }
  return result; // e.g. "aaabbc" → "3a2bc"
}

5 Keamanan Data

5.1 Enkripsi XOR + Base64 (storage.js)

Data skor dienkripsi menggunakan XOR cipher dengan kunci 'EnglishKu2024' sebelum di-encode Base64 dan disimpan. Data tidak bisa dibaca langsung dari DevTools.

// storage.js – XOR cipher
function encode(str) {
  const key = 'EnglishKu2024';
  let result = '';
  for (let i = 0; i < str.length; i++) {
    result += String.fromCharCode(
      str.charCodeAt(i) ^ key.charCodeAt(i % key.length)
    );
  }
  return btoa(result); // → Base64
}

5.2 Integritas Data – Checksum

Setiap kali data disimpan, dihitung checksum hash. Saat data dimuat, checksum diverifikasi. Jika tidak cocok (data dimanipulasi), data direset.

// storage.js – Djb2-variant hash checksum
_checksum(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash = hash & hash; // Convert to 32-bit int
  }
  return hash.toString(16);
}
